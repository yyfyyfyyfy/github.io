Python爬虫爬补天（重写)

发表于 2016-08-03   |  

上次写的爬虫爬取补天的厂商列表，如果厂商列表中只有厂商名而没有url是就会报错，而且for循环对于补天的厂商列表来说有些死板，因为他的厂商时刻在增加，所以把for循环改为while循环。

代码如下：

import requests
from lxml import etree
import os

def Save_File(messageList):
    path = '补天厂商列表'
    if not os.path.exists(path):
        os.mkdir(path)

    filepath = '厂商列表.txt'
    new_path = os.path.join(path, filepath)
    with open(new_path, 'a+', encoding = 'utf8') as f:
        for x in messageList:
            f.write('%s   %s\n'% (x[0][0], x[1][0]))    

def load_message(page_message):
    dom = etree.HTML(page_message)
    i = 2
    LM_messageList = []
    while True:
        index = str(i)
        new_xpath = '//table/tr[' + index + ']/td[1]/a/text()'
        title = dom.xpath(new_xpath)
        new_url_xpath = '//table/tr[' + index + ']/td[2]/text()'
        url = dom.xpath(new_url_xpath)
        if not url:
            url = ['URL丢失']

        if not title:
            break
        LM_messageList.append((title, url))
        i += 1
    return LM_messageList

def Spider(file_URL, last_message):
    print(file_URL)
    S_page_message = requests.get(file_URL).content.decode('utf8')
    S_messageList = load_message(S_page_message)
    a = S_messageList[0][0] == last_message[0]
    if not S_messageList[0][0] == last_message:
        Save_File(S_messageList)

    return S_messageList[0][0], a

if __name__ == '__main__':
    new_url = 'https://butian.360.cn/company/lists'
    t = Spider(new_url, ('a', 'b'))
    i = 2
    while True:
        P_new_url = new_url +'/page/' + str(i)
        t = Spider(P_new_url, t)
        if  t[1]:
            break
        i += 1
    print('finish')

重写后的代码取消了文件夹，只有一个补天列表文件，会自动判断有无url，且无论厂商有多少页都可以爬到，更加全面。
